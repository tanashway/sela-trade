# SMTP Configuration for Contact Form
# Copy this file to .env.local and fill in your SMTP credentials
# These settings are used to send contact form emails to info@selatrade.com

# SMTP Server Settings
SMTP_HOST=smtp.gmail.com
SMTP_PORT=587
SMTP_SECURE=false

# SMTP Authentication
SMTP_USER=your-email@gmail.com
SMTP_PASS=your-app-password

# From Address (optional - defaults to SMTP_USER)
SMTP_FROM=noreply@selatrade.com

# For Gmail:
# 1. Enable 2-Factor Authentication in your Google Account
# 2. Go to https://myaccount.google.com/apppasswords
# 3. Generate an App Password for "Mail"
# 4. Use that 16-character password as SMTP_PASS

# For other providers:
# - Outlook/Office365: smtp.office365.com, port 587
# - Yahoo: smtp.mail.yahoo.com, port 587
# - Zoho: smtp.zoho.com, port 587
# - SendGrid: smtp.sendgrid.net, port 587
